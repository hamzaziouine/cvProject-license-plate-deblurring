% ---- 2: Problem & Goal ----
\begin{frame}{Problem \& Goal}
\textbf{Problem:} License plates in surveillance footage are blurred (motion, defocus, atmospheric).

\vspace{0.2cm}
\textbf{Degradation model:} \quad $y = k \ast x + n$

\vspace{0.3cm}
\textbf{Goal:} Build a classical pipeline to recover $x$ from $y$:

\vspace{0.2cm}
\begin{center}
\small
\fbox{Blurry Image} $\rightarrow$
\fbox{FFT / PSF Est.} $\rightarrow$
\fbox{Deconvolution} $\rightarrow$
\fbox{TV Denoising} $\rightarrow$
\fbox{OCR}
\end{center}
\end{frame}

% ---- 4: PSF Estimation ----
\begin{frame}{Blind PSF Estimation via FFT}
\begin{enumerate}\setlength{\itemsep}{2pt}
  \item Compute magnitude spectrum $|F(u,v)|$ of blurry image
  \item \textbf{Angle:} angular sweep --- minimum energy direction = blur angle
  \item \textbf{Length:} null spacing along dark band --- $L \approx 1/\Delta f$
\end{enumerate}

% ---- 5: Deconvolution Methods ----
\begin{frame}{5 Deconvolution Methods}
\small
\begin{enumerate}\setlength{\itemsep}{4pt}
  \item \textbf{Inverse Filter:} $\hat{X} = Y / K$ \hfill \textit{(baseline, amplifies noise)}

  \item \textbf{Wiener Filter:}
  $\hat{X} = \frac{K^*}{|K|^2 + S_n/S_x} \cdot Y$ \hfill \textit{(optimal, known PSF)}

  \item \textbf{Unsupervised Wiener:} auto noise-to-signal estimation \hfill \textit{(no tuning)}

  \item \textbf{Richardson--Lucy:}
  $\hat{x}^{(t+1)} = \hat{x}^{(t)} \cdot \bigl( k^T \ast \frac{y}{k \ast \hat{x}^{(t)}} \bigr)$ \hfill \textit{(iterative, Poisson)}

  \item \textbf{CLS:}
  $\hat{X} = \frac{K^*}{|K|^2 + \gamma |P|^2} \cdot Y$ \hfill \textit{(Laplacian regularisation)}
\end{enumerate}
\end{frame}

\begin{column}{0.5\textwidth}
\textbf{Key findings:}
\begin{itemize}
  \item All below baseline --- PSF estimation error is the bottleneck
  \item With \textbf{oracle PSF}, Wiener gains \textbf{+2.2\,dB} --- methods work, estimation doesn't
  \item Wiener is best overall performer
\end{itemize}
\end{column}

% ---- 9: Limitations & Future Work ----
\begin{frame}{Limitations \& Future Work}
\textbf{Limitations:}
\begin{itemize}
  \item Blind PSF estimation is the primary bottleneck
  \item Synthetic blur $\neq$ real-world degradation
  \item Spatially varying blur not addressed
\end{itemize}

\vspace{0.5cm}
\textbf{Future work:}
\begin{itemize}
  \item Multi-scale / adaptive PSF estimation
  \item Real surveillance footage evaluation
  \item Automatic blur-type classification
\end{itemize}
