\section{Introduction}

% ---- Slide 3: Problem Statement ----
\begin{frame}{Problem Statement}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Challenge:} License plates in surveillance footage are often blurry due to:
\begin{itemize}
  \item Motion blur (moving vehicles)
  \item Defocus blur (camera optics)
  \item Atmospheric degradation
\end{itemize}

\vspace{0.5cm}
\textbf{Impact:} OCR systems fail on blurry plates
\begin{itemize}
  \item Only \textbf{33\%} of blurry plates are readable by OCR
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\centering
\textbf{Degradation Model:}
\vspace{0.3cm}
$$y = k \ast x + n$$
\begin{itemize}
  \item $x$: sharp image
  \item $k$: blur kernel (PSF)
  \item $n$: noise
  \item $y$: observed blurry image
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\section{Methodology}

% ---- Slide 6: Blur Estimation via FFT ----
\begin{frame}{Blur Estimation via FFT}
\textbf{Blind PSF estimation from the frequency domain:}

\vspace{0.3cm}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Motion blur angle estimation:}
\begin{enumerate}
  \item Compute $|F(u,v)|$ of the blurry image
  \item The motion blur PSF creates a \textbf{dark band} perpendicular to the motion direction
  \item \textbf{Angular sweep:} integrate energy along radial lines at each angle $\theta \in [0^\circ, 180^\circ)$
  \item The angle with minimum energy = blur direction
\end{enumerate}
\end{column}
\begin{column}{0.4\textwidth}
\textbf{Motion blur length estimation:}
\begin{enumerate}
  \item Extract 1D profile along the dark band direction
  \item Count \textbf{zero-crossings} or measure spacing of periodic nulls
  \item Null spacing $\Delta f \approx 1/L$, where $L$ = blur length in pixels
\end{enumerate}

\vspace{0.3cm}
\textbf{Result:} Estimated PSF $\hat{k}$ for deconvolution
\end{column}
\end{columns}
\end{frame}

% ---- Slide 7: Classical Deconvolution Methods ----
\begin{frame}{Classical Deconvolution Methods}
\textbf{Progression of increasingly robust approaches:}

\vspace{0.3cm}
\begin{enumerate}
  \item \textbf{Inverse Filter} (baseline): $\hat{X} = Y / K$ \hfill \textit{amplifies noise}

  \item \textbf{Wiener Filter} (optimal for known PSF + noise):
  $$\hat{X}(u,v) = \frac{K^*(u,v)}{|K(u,v)|^2 + S_n/S_x} \cdot Y(u,v)$$

  \item \textbf{Unsupervised Wiener}: automatic noise-to-signal ratio estimation

  \item \textbf{Richardson--Lucy}: iterative Bayesian deconvolution (Poisson noise model)

  \item \textbf{Constrained Least Squares (CLS)}: Laplacian regularisation for edge preservation
\end{enumerate}

\vspace{0.3cm}
\small{All methods take the estimated PSF $\hat{k}$ as input; differ in noise handling and regularisation strategy.}
\end{frame}

% ---- Slide 8: Richardson--Lucy Details ----
\begin{frame}{Richardson--Lucy Deconvolution}
\textbf{Iterative update rule (Bayesian, Poisson noise model):}
$$\hat{x}^{(t+1)} = \hat{x}^{(t)} \cdot \left( k^T \ast \frac{y}{k \ast \hat{x}^{(t)}} \right)$$

\vspace{0.3cm}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Properties:}
\begin{itemize}
  \item Non-negativity preserved at each iteration
  \item Converges to maximum likelihood estimate
  \item No explicit noise model parameter needed
  \item Robust to moderate PSF mismatch
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{Practical considerations:}
\begin{itemize}
  \item Number of iterations controls regularisation
  \item Too few $\rightarrow$ under-deconvolved
  \item Too many $\rightarrow$ noise amplification (ringing)
  \item Typical range: 10--50 iterations
\end{itemize}
\end{column}
\end{columns}
\end{frame}

% ---- Slide 9: CLS Filter Details ----
\begin{frame}{Constrained Least Squares (CLS) Filter}
\textbf{Minimize:} $\| C \hat{x} \|^2$ \quad subject to \quad $\| y - k \ast \hat{x} \|^2 = \| n \|^2$

\vspace{0.3cm}
\textbf{Frequency domain solution:}
$$\hat{X}(u,v) = \frac{K^*(u,v)}{|K(u,v)|^2 + \gamma |P(u,v)|^2} \cdot Y(u,v)$$

where $P(u,v)$ is the Laplacian operator and $\gamma$ is the regularisation parameter.

\vspace{0.3cm}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Advantages:}
\begin{itemize}
  \item Laplacian penalty preserves edges
  \item Single parameter $\gamma$ to tune
  \item Closed-form solution (fast)
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\textbf{Comparison to Wiener:}
\begin{itemize}
  \item Wiener: penalizes deviation from signal statistics
  \item CLS: penalizes non-smooth solutions
  \item CLS better when noise statistics are unknown
\end{itemize}
\end{column}
\end{columns}
\end{frame}

% ---- Slide 10: Post-Processing ----
\begin{frame}{Post-Processing: Total Variation Denoising}
\textbf{Problem:} Deconvolution introduces ringing artifacts near edges.

\vspace{0.3cm}
\textbf{Solution:} Total Variation (TV) denoising as post-processing step:
$$\min_{\hat{x}} \left\{ \frac{1}{2} \| \hat{x} - x_{\text{deconv}} \|^2 + \lambda \cdot \text{TV}(\hat{x}) \right\}$$

where $\text{TV}(\hat{x}) = \sum_{i,j} \sqrt{(\nabla_h \hat{x})^2 + (\nabla_v \hat{x})^2}$

\vspace{0.3cm}
\begin{itemize}
  \item Preserves sharp edges while suppressing oscillations
  \item Weight $\lambda$ balances fidelity vs.\ smoothness
  \item Applied after each deconvolution method
  \item Particularly effective for reducing Gibbs-like ringing from frequency-domain methods
\end{itemize}
\end{frame}

% ---- Slide 15: Convolution Connection (Lab 1) ----
\begin{frame}{Convolution: The Unifying Concept (Lab 1)}
\textbf{Blur is convolution. All our methods are deconvolution.}

\vspace{0.3cm}
\begin{enumerate}
  \item \textbf{Blur generation:} $y = k \ast x$ (spatial convolution with PSF)
  \item \textbf{Wiener / CLS filters:} Deconvolution via frequency domain division
    \begin{itemize}
      \item Multiplication in frequency domain = convolution in spatial domain
    \end{itemize}
  \item \textbf{Richardson--Lucy:} Iterative deconvolution in the spatial domain
    \begin{itemize}
      \item Each iteration involves convolution with $k$ and correlation with $k^T$
    \end{itemize}
\end{enumerate}

\vspace{0.3cm}
\textbf{Every method in this project is rooted in Lab 1:}
\begin{itemize}
  \item Frequency-domain methods: convolution $\leftrightarrow$ multiplication (Fourier theorem)
  \item Spatial-domain methods: direct convolution with PSF and its transpose
  \item PSF estimation: analyzing convolution effects in the Fourier domain
\end{itemize}
\end{frame}

% ---- Slide 17: Limitations & Future Work ----
\begin{frame}{Limitations \& Future Work}
\textbf{Limitations:}
\begin{itemize}
  \item PSF mismatch remains the primary challenge for blind deconvolution
  \item Synthetic-to-real gap: controlled blur differs from real-world degradation
  \item Spatially varying blur (e.g., radial motion) not addressed
  \item Parameter tuning ($\gamma$, iterations, $\lambda$) is dataset-dependent
\end{itemize}

\vspace{0.5cm}
\textbf{Future directions:}
\begin{itemize}
  \item Multi-scale PSF estimation for spatially varying blur
  \item Adaptive parameter selection (automatic $\gamma$ and iteration count)
  \item Multi-kernel PSF estimation with automatic blur-type classification
  \item Evaluation on real surveillance footage
  \item Moroccan license plate format specialization
\end{itemize}
